const fs = require('fs');
 
function writeFile(data, path) {
  return new Promise(function(resolve, reject) {
    let wstream = fs.createWriteStream(path);
    wstream.on('finish', () => resolve(path))
    wstream.on('error', (err) => reject(err))
    wstream.write(data)
    wstream.end()
  });
}
